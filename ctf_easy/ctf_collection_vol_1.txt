# Define the hex strings
s1 = "44585d6b2368737c65252166234f20626d"

# Convert hex strings to bytes
b1 = bytes.fromhex(s1)

# Ensure the strings are the same length
if len(b1) != len(b2):
    raise ValueError("Inputs must be the same length.")
